
            heapSort: function void (out arr:array [100] of integer, n:integer){
                // Build heap (rearrange array)
                for (i = N / 2 - 1, i >= 0, i-1)
                    heapify(arr, N, i);
            
                // One by one extract an element
                // from heap
                for (i = N - 1, i > 0, i-1) {
            
                    // Move current root to end
                    swap(arr[0], arr[i]);
            
                    // call max heapify on the reduced heap
                    heapify(arr, i, 0);
                }
            }
